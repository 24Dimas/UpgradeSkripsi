function doPost(e) {
  // ===== LOGIN ADMIN via POST (aman, tidak lewat URL) =====
  if (e.parameter.action === "login") {
    return handleLogin(e);
  }

  // ===== KIRIM EMAIL DARI DASHBOARD (POST) =====
  if (e.parameter.action === "sendEmail") {
    return handleSendEmail(e);
  }

  // ===== FORM PENDAFTARAN =====
  try {
        var data = e.parameter.fileContent;
        var filename = e.parameter.filename;
        var email = e.parameter.email;
        var nama = e.parameter.nama;
        var revisiRef = (e.parameter.revisi_ref || "").trim(); // <-- revisi support

        var refNum;
        var isRevisi = false;

        // Cek apakah ini revisi: revisi_ref ada DAN row dengan ref tsb ada DAN status "Ditolak"
        if (revisiRef) {
          var checkRow = findRowByRefForRevisi_(revisiRef);
          if (checkRow > 0) {
            isRevisi = true;
            refNum = revisiRef; // tetap pakai nomor ref lama
          }
        }
        if (!refNum) {
          // Generate nomor referensi baru
          refNum = "KBP-" + Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyyMMdd-HHmm");
        }

        var result = uploadFileToGoogleDrive(data, filename, nama, email, e, refNum);

        // Kirim email konfirmasi (template beda untuk revisi)
        kirimEmailKonfirmasi(e, refNum, isRevisi);

        return ContentService
            .createTextOutput(
                JSON.stringify({"Anda berhasil mendaftar": isRevisi ? "Data revisi telah terkirim" : "Data telah terkirim", "ref": refNum, "isRevisi": isRevisi})
            )
            .setMimeType(ContentService.MimeType.JSON);
    } catch (error) {
        Logger.log(error);
        return ContentService
            .createTextOutput(JSON.stringify({"result": "error", "error": error}))
            .setMimeType(ContentService.MimeType.JSON);
    }
}

// Helper: cari row dgn ref number, hanya jika status "Ditolak"
function findRowByRefForRevisi_(refNumber) {
  try {
    var doc = SpreadsheetApp.openById('1p5vDE8-dQbLxQbaxeE9nknDQ0W53QBqibE9qxmOa-0Q');
    var sheet = doc.getSheetByName('Data');
    var values = sheet.getDataRange().getValues();
    var headers = values[0].map(function(h){ return h.toString().toLowerCase().trim(); });
    var refCol = headers.indexOf('ref_number');
    var statusCol = headers.indexOf('status');
    if (refCol < 0 || statusCol < 0) return -1;
    for (var i = 1; i < values.length; i++) {
      var ref = values[i][refCol] ? values[i][refCol].toString().trim() : "";
      var status = values[i][statusCol] ? values[i][statusCol].toString().trim() : "";
      if (ref === refNumber && status.toLowerCase() === "ditolak") {
        return i + 1; // row index 1-based
      }
    }
  } catch (err) { Logger.log("findRowByRefForRevisi_ error: " + err); }
  return -1;
}

function record_data(e, fileUrl, refNum) {
    try {
        var doc = SpreadsheetApp.openById('1p5vDE8-dQbLxQbaxeE9nknDQ0W53QBqibE9qxmOa-0Q');
        var sheet = doc.getSheetByName('Data');
        var timestamp = new Date();
        var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];

        // Cek apakah ini revisi
        var revisiRef = (e.parameter.revisi_ref || "").trim();
        var targetRow = -1;
        var isRevisi = false;
        if (revisiRef) {
          targetRow = findRowByRefForRevisi_(revisiRef);
          if (targetRow > 0) isRevisi = true;
        }

        var row = [
            Utilities.formatDate(timestamp, Session.getScriptTimeZone(), "yyyy-MM-dd HH:mm:ss")
        ];

        for (var i = 1; i < headers.length; i++) {
            var h = headers[i].toString().toLowerCase().trim();
            if (h == "file")               row.push(fileUrl);
            else if (h == "status")        row.push("Pending"); // revisi: status balik ke Pending
            else if (h == "ref_number")    row.push(refNum || "");
            else if (h == "revisi_count") {
              // Increment counter kalau revisi
              if (isRevisi) {
                var currentVal = sheet.getRange(targetRow, i + 1).getValue();
                var count = parseInt(currentVal) || 0;
                row.push(count + 1);
              } else {
                row.push(0);
              }
            }
            else                           row.push(e.parameter[headers[i]] || "");
        }

        if (isRevisi && targetRow > 0) {
          // UPDATE row yang sudah ada
          sheet.getRange(targetRow, 1, 1, row.length).setValues([row]);
        } else {
          // APPEND row baru
          var nextRow = sheet.getLastRow() + 1;
          sheet.getRange(nextRow, 1, 1, row.length).setValues([row]);
        }
    } catch (error) {
        Logger.log(error);
    }
}

function uploadFileToGoogleDrive(data, file, nama, email, e, refNum) {
    try {
        var folder = DriveApp.getFoldersByName("REGISTRASI").hasNext() ? DriveApp.getFoldersByName("REGISTRASI").next() : DriveApp.createFolder("REGISTRASI");

        var bytes = Utilities.base64Decode(data.split(',')[1]);
        var blob = Utilities.newBlob(bytes, data.substring(5, data.indexOf(';')), file);
        var file = folder.createFile(blob);
        
        record_data(e, file.getUrl(), refNum);
        return file.getUrl();
    } catch (error) {
        return null;
    }
}

function formatTanggalIndonesia(date) {
  if (!date) return "";

  const bulanIndo = [
    "Januari", "Februari", "Maret", "April",
    "Mei", "Juni", "Juli", "Agustus",
    "September", "Oktober", "November", "Desember"
  ];

  const d = new Date(date);
  const tanggal = d.getDate();
  const bulan = bulanIndo[d.getMonth()];
  const tahun = d.getFullYear();

  return `${tanggal} ${bulan} ${tahun}`;
}

function doGet(e) {
  const action = e.parameter.action;
 
  if (action === "login") {
    return handleLogin(e);
  } else if (action === "getData") {
    return handleGetData();
  } else if (action === "updateStatus") {
    return handleUpdateStatus(e);
  } else if (action === "generateSurat") {
    return handleGenerateSurat(e);
  } else if (action === "checkStatus") {
    return handleCheckStatus(e);
  } else if (action === "getDataByRef") {
    return handleGetDataByRef(e);
  }
 
  return ContentService
    .createTextOutput(JSON.stringify({ error: "Action tidak dikenal" }))
    .setMimeType(ContentService.MimeType.JSON);
}

// ===== GET DATA BY REF (untuk revisi auto-fill) =====
// Return data row by ref_number, HANYA jika status "Ditolak" (security)
function handleGetDataByRef(e) {
  try {
    const ref = (e.parameter.ref || "").trim();
    if (!ref) return jsonResponse({ found: false, error: "Nomor referensi tidak ada" });

    const doc = SpreadsheetApp.openById('1p5vDE8-dQbLxQbaxeE9nknDQ0W53QBqibE9qxmOa-0Q');
    const sheet = doc.getSheetByName('Data');
    const values = sheet.getDataRange().getValues();
    const headers = values[0].map(function(h){ return h.toString().toLowerCase().trim(); });
    const refCol = headers.indexOf('ref_number');
    const statusCol = headers.indexOf('status');
    if (refCol < 0 || statusCol < 0) return jsonResponse({ found: false, error: "Kolom tidak ditemukan" });

    for (var i = 1; i < values.length; i++) {
      var rowRef = values[i][refCol] ? values[i][refCol].toString().trim() : "";
      var rowStatus = values[i][statusCol] ? values[i][statusCol].toString().trim() : "";
      if (rowRef === ref) {
        // Hanya boleh revisi kalau status "Ditolak"
        if (rowStatus.toLowerCase() !== "ditolak") {
          return jsonResponse({ found: false, error: "Pengajuan ini tidak dapat direvisi (status: " + rowStatus + ")" });
        }
        // Build data object
        var row = {};
        headers.forEach(function(h, idx) {
          row[h] = values[i][idx] ? values[i][idx].toString() : "";
        });
        return jsonResponse({ found: true, data: row });
      }
    }
    return jsonResponse({ found: false, error: "Nomor referensi tidak ditemukan" });
  } catch (err) {
    return jsonResponse({ found: false, error: err.toString() });
  }
}
 
function handleLogin(e) {
  try {
    const username       = e.parameter.username;
    const passwordHashed = e.parameter.password; // sudah di-hash SHA-256 dari browser
 
    const doc = SpreadsheetApp.openById('1p5vDE8-dQbLxQbaxeE9nknDQ0W53QBqibE9qxmOa-0Q');
    const sheet = doc.getSheetByName('Admin');
 
    if (!sheet) {
      return jsonResponse({ success: false, message: "Sheet Admin tidak ditemukan" });
    }
 
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      const storedUser = data[i][0] ? data[i][0].toString().trim() : "";
      const storedHash = data[i][1] ? data[i][1].toString().trim() : "";
      if (storedUser === username && storedHash === passwordHashed) {
        return jsonResponse({ success: true });
      }
    }
 
    return jsonResponse({ success: false });
  } catch (err) {
    return jsonResponse({ success: false, error: err.toString() });
  }
}
 
// ===== GET DATA =====
function handleGetData() {
  try {
    const doc = SpreadsheetApp.openById('1p5vDE8-dQbLxQbaxeE9nknDQ0W53QBqibE9qxmOa-0Q');
    const sheet = doc.getSheetByName('Data');
    const data = sheet.getDataRange().getValues();
 
    if (data.length < 2) return jsonResponse({ data: [] });
 
    const headers = data[0].map(h => h.toString().toLowerCase().trim());
    const rows = [];
 
    for (let i = 1; i < data.length; i++) {
      const row = {};
      headers.forEach((h, idx) => { row[h] = data[i][idx] ? data[i][idx].toString() : ""; });
      row.rowIndex = i + 1;
      rows.push(row);
    }
 
    return jsonResponse({ data: rows });
  } catch (err) {
    return jsonResponse({ data: [], error: err.toString() });
  }
}
 
// ===== UPDATE STATUS =====
function handleUpdateStatus(e) {
  try {
    const rowIndex = parseInt(e.parameter.rowIndex);
    const status = e.parameter.status;
 
    const doc = SpreadsheetApp.openById('1p5vDE8-dQbLxQbaxeE9nknDQ0W53QBqibE9qxmOa-0Q');
    const sheet = doc.getSheetByName('Data');
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    const colStatus = headers.findIndex(h => h.toString().toLowerCase().trim() === "status") + 1;
 
    if (!colStatus) {
      return jsonResponse({ success: false, message: "Kolom status tidak ditemukan" });
    }
 
    sheet.getRange(rowIndex, colStatus).setValue(status);
    return jsonResponse({ success: true });
  } catch (err) {
    return jsonResponse({ success: false, error: err.toString() });
  }
}

// ===== GENERATE SURAT DARI DASHBOARD =====
function handleGenerateSurat(e) {
  try {
    const rowIndex = parseInt(e.parameter.rowIndex);
 
    const SPREADSHEET_ID = "1p5vDE8-dQbLxQbaxeE9nknDQ0W53QBqibE9qxmOa-0Q";
    const TEMPLATE_ID    = "1Q3hlI1prise-DRQLj-TXxvFlD-MZR_5nlxIDRpZGXuQ";
    const FOLDER_ID      = "1p4j741DW88_6w4nLcobhnOwdTNiAgbuR";
 
    const ss      = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet   = ss.getSheetByName("Data");
    const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(h => h.toString().toLowerCase().trim());
    const rowData = sheet.getRange(rowIndex, 1, 1, sheet.getLastColumn()).getValues()[0];
 
    const row = {};
    headers.forEach((h, idx) => { row[h] = rowData[idx] ? rowData[idx].toString() : ""; });
 
    const copy = DriveApp.getFileById(TEMPLATE_ID).makeCopy(
      `Surat Penelitian - ${row.nama || "Tanpa Nama"}`,
      DriveApp.getFolderById(FOLDER_ID)
    );
 
    const doc  = DocumentApp.openById(copy.getId());
    const body = doc.getBody();
 
    const fields = ["atasnama","fakultas","nama_lembaga","nomor_surat","perihal",
      "nama","alamat","rukun_tetangga_rt","rukun_warga_rw","kecamatan","kabupaten",
      "pekerjaan","email","nomor_hp","proposal","lokasi","bidang","penanggung_jawab","anggota_peneliti","file"];
 
    fields.forEach(f => body.replaceText(`{{${f}}}`, row[f] || ""));
 
    body.replaceText("{{tgl_surat}}",      formatTanggalIndonesia(row.tgl_surat));
    body.replaceText("{{tanggal_mulai}}",  formatTanggalIndonesia(row.tanggal_mulai));
    body.replaceText("{{tanggal_selesai}}",formatTanggalIndonesia(row.tanggal_selesai));
 
    doc.saveAndClose();
 
    const colLink = headers.findIndex(h => h === "link_surat") + 1;
    if (colLink) sheet.getRange(rowIndex, colLink).setValue(copy.getUrl());
 
    return jsonResponse({ success: true, link: copy.getUrl() });
  } catch (err) {
    return jsonResponse({ success: false, error: err.toString() });
  }
}
 
// ===== HELPER =====
function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

function handleCheckStatus(e) {
  try {
    const ref = (e.parameter.ref || "").trim().toUpperCase();
    if (!ref) return jsonResponse({ found: false });

    const doc   = SpreadsheetApp.openById("1p5vDE8-dQbLxQbaxeE9nknDQ0W53QBqibE9qxmOa-0Q");
    const sheet = doc.getSheetByName("Data");
    const data  = sheet.getDataRange().getValues();

    if (data.length < 2) return jsonResponse({ found: false });

    const headers = data[0].map(h => h.toString().toLowerCase().trim());
    const colRef  = headers.indexOf("ref_number");

    if (colRef === -1) return jsonResponse({ found: false, error: "Kolom ref_number tidak ada di sheet" });

    for (let i = 1; i < data.length; i++) {
      const rowRef = (data[i][colRef] || "").toString().trim().toUpperCase();
      if (rowRef === ref) {
        const row = {};
        headers.forEach((h, idx) => { row[h] = data[i][idx] ? data[i][idx].toString() : ""; });
        return jsonResponse({ found: true, row });
      }
    }

    return jsonResponse({ found: false });
  } catch (err) {
    return jsonResponse({ found: false, error: err.toString() });
  }
}
